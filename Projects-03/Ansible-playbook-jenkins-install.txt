---
- name: install jenkins
  hosts: 172.31.41.165    
  become: true
  gather_facts: no

  tasks:

  - name: install java
    yum:
      name: java-17-amazon-corretto
      state: installed

  - name: install wget
    yum:
      name: wget
      state: installed

  - name: install jenkins
    get_url:
      url: https://pkg.jenkins.io/redhat-stable/jenkins.repo
      dest: /etc/yum.repos.d/jenkins.repo

  - name: add key
    rpm_key:
      state: present
      key: https://pkg.jenkins.io/redhat-stable/jenkins.io-2023.key

  - name: install jenkins
    yum:
      name: jenkins
      state: installed

  - name: start jenkins
    service:
      name: jenkins
      state: started
      enabled: yes
